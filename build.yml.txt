name: CI

on:
  push:
    branches:
      - main

jobs:
  build:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout repository
        uses: actions/checkout@v4

      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: "3.10"

      - name: Install Android Build Dependencies
        run: |
          sudo apt update
          sudo apt install -y git zip unzip openjdk-17-jdk autoconf libtool pkg-config zlib1g-dev libncurses5-dev libgdbm-dev libssl-dev libffi-dev libsqlite3-dev tk-dev libgdbm-compat-dev libsdl2-dev libsdl2-image-dev libsdl2-mixer-dev libsdl2-ttf-dev
          sudo apt install -y libjpeg-dev jpegoptim jq libwebp-dev libpng-dev

      - name: Install Buildozer and Dependencies
        run: |
          python -m pip install --upgrade pip
          pip install buildozer cython virtualenv sh

      - name: Prepare Spec and Build APK
        run: |
          # 1. NO borres tu buildozer.spec. En su lugar, asegúrate de que acepte la licencia automáticamente.
          # Esto descomenta y activa android.accept_sdk_license = True en tu archivo real.
          sed -i 's/#\s*android.accept_sdk_license\s*=\s*False/android.accept_sdk_license = True/' buildozer.spec
          
          # 2. Forzar que use la API 33 y la NDK estable si no estaban descomentadas (opcional pero seguro)
          sed -i 's/#\s*android.api\s*=\s*33/android.api = 33/' buildozer.spec
          
          # 3. Limpieza profunda obligatoria para entornos CI
          buildozer android clean
          
          # 4. Compilar directamente (sin interactividad)
          buildozer android debug

      - name: Upload APK
        uses: actions/upload-artifact@v4
        with:
          name: app-release-apk
          path: bin/*.apk
