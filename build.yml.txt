name: CI

on:
  push:
    branches:
      - main

jobs:
  build:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout repository
        uses: actions/checkout@v4

      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: "3.10"

      - name: Install Android Build Dependencies
        run: |
          sudo apt update
          sudo apt install -y git zip unzip openjdk-17-jdk autoconf libtool pkg-config zlib1g-dev libncurses5-dev libgdbm-dev libssl-dev libffi-dev libsqlite3-dev tk-dev libgdbm-compat-dev libsdl2-dev libsdl2-image-dev libsdl2-mixer-dev libsdl2-ttf-dev
          sudo apt install -y libjpeg-dev jpegoptim jq libwebp-dev libpng-dev

      - name: Install Buildozer and Dependencies
        run: |
          python -m pip install --upgrade pip
          pip install buildozer cython virtualenv sh

      - name: Generate and Build with Stable Config
        run: |
          # 1. Si no existe un buildozer.spec, creamos uno base. Si existe, lo borramos para evitar configuraciones corruptas.
          rm -f buildozer.spec
          buildozer init
          
          # 2. Reemplazar valores críticos con configuraciones ultra-estables para GitHub Actions
          sed -i 's/# android.ndk_version = .*/android.ndk_version = 26b/' buildozer.spec
          sed -i 's/android.ndk_version = .*/android.ndk_version = 26b/' buildozer.spec
          sed -i 's/android.api = .*/android.api = 33/' buildozer.spec
          sed -i 's/requirements = .*/requirements = python3, kivy==2.3.0, pillow, jpeg, png/' buildozer.spec
          
          # 3. Limpieza profunda del espacio de trabajo de compilación anterior
          buildozer android clean
          
          # 4. Compilar en modo debug de manera directa
          buildozer android debug

      - name: Upload APK
        uses: actions/upload-artifact@v4
        with:
          name: app-release-apk
          path: "**/*.apk"
