- name: Prepare Spec and Build APK
        env:
          # Esta variable fuerza la aceptación de licencias a nivel de SDK de Android
          ANDROID_擺C_LICENSE: "True"
          # Variable estándar para herramientas de Android en entornos CI
          AUTO_ACCEPT_AND_SDK_LICENSES: "true"
        run: |
          # 1. Asegurar la aceptación desde el buildozer.spec forzando la línea si no existe
          # Buscamos si existe la propiedad; si no, la añadimos al final de la sección [app]
          grep -q "android.accept_sdk_license" buildozer.spec && \
            sed -i 's/.*android.accept_sdk_license.*/android.accept_sdk_license = True/' buildozer.spec || \
            sed -i '/\[app\]/a android.accept_sdk_license = True' buildozer.spec
          
          # 2. Limpieza profunda del entorno para borrar residuos corruptos de compilaciones previas
          buildozer android clean
          
          # 3. Compilar de manera directa (sin 'yes |')
          buildozer android debug
